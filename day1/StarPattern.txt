using System;
					
public class Program
{
	public static void Main()
	{
		int Number;
		 Number = Convert.ToInt32(Console.ReadLine());
		 for( int i=1;i<=Number;i++)
		 {
		for(int j=1;j<=i;j++)
		{
 		Console.Write(" * ");
 }

 Console.WriteLine();
 }
 Console.ReadLine();
 }
 }
