using System;  
public class array
{  
    public static void Main() 
{
	int[,] arr = new int[2,2];
		Console.Write("Enter values");
		for(int row=0; row < 2; row++)
		{
				for(int col=0; col < 2; col++)
				{
					arr[row,col] = Convert.ToInt32(Console.ReadLine());
				}
		}
		for(int row=0; row<2;row++)
		{
			for(int col=0; col < 2; col++)
			{
				Console.Write(arr[row,col]+"\t");
			}
			Console.WriteLine();
		}
		Console.ReadLine();
	}
}
