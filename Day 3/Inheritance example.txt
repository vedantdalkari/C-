using System;

namespace ConsoleApp1
{
    public class Animal
    {
        public void eat()
        {
            Console.WriteLine("Dog can eat");
        }
    }
    public class Dog: Animal
    {
        public void bark()
        {
            Console.WriteLine("Dog can Bark");
        }
        static void Main(string[] args)
        {
            Dog d = new Dog();
            d.eat();
            d.bark();
            Console.ReadLine();
        }
    }
}
